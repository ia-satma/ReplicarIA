javascriptimport { useState, useCallback, useRef, useEffect } from 'react';

export function useChatMessages() {
  const [messages, setMessages] = useState([]);
  const [conversationHistory, setConversationHistory] = useState([]);
  const [isTyping, setIsTyping] = useState(false);
  const [isStreaming, setIsStreaming] = useState(false);
  const streamingMessageRef = useRef(null);
  const messagesEndRef = useRef(null);

  // Auto-scroll al final cuando hay nuevos mensajes
  useEffect(() => {
    messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' });
  }, [messages]);

  const addUserMessage = useCallback((content, attachments = []) => {
    const userMessage = {
      id: `user-${Date.now()}`,
      role: 'user',
      content,
      attachments,
      timestamp: new Date().toISOString(),
    };

    setMessages((prev) => [...prev, userMessage]);
    
    // Agregar al historial para la API
    setConversationHistory((prev) => [
      ...prev,
      { role: 'user', content }
    ]);

    return userMessage;
  }, []);

  const addBotMessage = useCallback((content, metadata = {}) => {
    const botMessage = {
      id: `bot-${Date.now()}`,
      role: 'assistant',
      content,
      timestamp: new Date().toISOString(),
      ...metadata, // agent, agentName, suggestions, etc.
    };

    setMessages((prev) => [...prev, botMessage]);
    
    // Agregar al historial para la API
    setConversationHistory((prev) => [
      ...prev,
      { role: 'assistant', content }
    ]);

    return botMessage;
  }, []);

  const addSystemMessage = useCallback((content, type = 'info') => {
    const systemMessage = {
      id: `system-${Date.now()}`,
      role: 'system',
      content,
      type, // 'info', 'error', 'success', 'warning'
      timestamp: new Date().toISOString(),
    };

    setMessages((prev) => [...prev, systemMessage]);
    return systemMessage;
  }, []);

  const startStreaming = useCallback((metadata = {}) => {
    setIsStreaming(true);
    
    const streamingMessage = {
      id: `streaming-${Date.now()}`,
      role: 'assistant',
      content: '',
      isStreaming: true,
      timestamp: new Date().toISOString(),
      ...metadata,
    };

    streamingMessageRef.current = streamingMessage.id;
    setMessages((prev) => [...prev, streamingMessage]);
    
    return streamingMessage.id;
  }, []);

  const updateStreamingMessage = useCallback((chunk) => {
    if (!streamingMessageRef.current) return;

    setMessages((prev) =>
      prev.map((msg) =>
        msg.id === streamingMessageRef.current
          ? { ...msg, content: msg.content + chunk }
          : msg
      )
    );
  }, []);

  const finalizeStreamingMessage = useCallback((finalContent = null, metadata = {}) => {
    if (!streamingMessageRef.current) return;

    setMessages((prev) =>
      prev.map((msg) =>
        msg.id === streamingMessageRef.current
          ? {
              ...msg,
              content: finalContent || msg.content,
              isStreaming: false,
              ...metadata,
            }
          : msg
      )
    );

    // Agregar al historial
    const finalMessage = messages.find(m => m.id === streamingMessageRef.current);
    if (finalMessage) {
      setConversationHistory((prev) => [
        ...prev,
        { role: 'assistant', content: finalContent || finalMessage.content }
      ]);
    }

    streamingMessageRef.current = null;
    setIsStreaming(false);
  }, [messages]);

  const cancelStreaming = useCallback(() => {
    if (!streamingMessageRef.current) return;

    // Remover mensaje de streaming
    setMessages((prev) => prev.filter((msg) => msg.id !== streamingMessageRef.current));
    
    streamingMessageRef.current = null;
    setIsStreaming(false);
  }, []);

  const setTyping = useCallback((typing, delay = 0) => {
    if (delay > 0) {
      setTimeout(() => setIsTyping(typing), delay);
    } else {
      setIsTyping(typing);
    }
  }, []);

  const clearMessages = useCallback(() => {
    setMessages([]);
    setConversationHistory([]);
  }, []);

  const removeMessage = useCallback((messageId) => {
    setMessages((prev) => prev.filter((msg) => msg.id !== messageId));
  }, []);

  const updateMessage = useCallback((messageId, updates) => {
    setMessages((prev) =>
      prev.map((msg) =>
        msg.id === messageId ? { ...msg, ...updates } : msg
      )
    );
  }, []);

  return {
    // State
    messages,
    conversationHistory,
    isTyping,
    isStreaming,
    messagesEndRef,

    // Actions
    addUserMessage,
    addBotMessage,
    addSystemMessage,
    startStreaming,
    updateStreamingMessage,
    finalizeStreamingMessage,
    cancelStreaming,
    setTyping,
    clearMessages,
    removeMessage,
    updateMessage,
  };
}

javascriptimport { useState, useCallback } from 'react';
import { onboardingApi } from '../../../api/onboarding';

const ALLOWED_EXTENSIONS = ['.pdf', '.docx', '.xlsx', '.txt', '.xml', '.png', '.jpg', '.jpeg'];
const MAX_FILE_SIZE = 50 * 1024 * 1024; // 50MB

export function useFileUpload(projectId, onAnalysisComplete) {
  const [uploadedFiles, setUploadedFiles] = useState([]);
  const [additionalFiles, setAdditionalFiles] = useState([]);
  const [isProcessingFile, setIsProcessingFile] = useState(false);
  const [isUploadingDocuments, setIsUploadingDocuments] = useState(false);
  const [fileAnalysisResults, setFileAnalysisResults] = useState([]);
  const [extractedDocumentContents, setExtractedDocumentContents] = useState([]);
  const [uploadProgress, setUploadProgress] = useState({});
  const [error, setError] = useState(null);

  const validateFile = useCallback((file) => {
    const extension = '.' + file.name.split('.').pop().toLowerCase();
    
    if (!ALLOWED_EXTENSIONS.includes(extension)) {
      return {
        valid: false,
        error: `Tipo de archivo no permitido: ${extension}. Permitidos: ${ALLOWED_EXTENSIONS.join(', ')}`
      };
    }

    if (file.size > MAX_FILE_SIZE) {
      return {
        valid: false,
        error: `Archivo muy grande: ${(file.size / 1024 / 1024).toFixed(2)}MB. M√°ximo: 50MB`
      };
    }

    return { valid: true };
  }, []);

  const handleFileSelect = useCallback(async (files, options = {}) => {
    const { analyzeWithAI = true, categoria = null } = options;
    
    setError(null);
    const validFiles = [];
    const errors = [];

    // Validar archivos
    for (const file of files) {
      const validation = validateFile(file);
      if (validation.valid) {
        validFiles.push({
          id: `file-${Date.now()}-${Math.random().toString(36).substr(2, 9)}`,
          file,
          name: file.name,
          size: file.size,
          type: file.type,
          status: 'pending',
          progress: 0,
        });
      } else {
        errors.push({ file: file.name, error: validation.error });
      }
    }

    if (errors.length > 0) {
      setError(errors.map(e => `${e.file}: ${e.error}`).join('\n'));
    }

    if (validFiles.length === 0) return [];

    setUploadedFiles((prev) => [...prev, ...validFiles]);

    // Analizar con IA si est√° habilitado
    if (analyzeWithAI) {
      await analyzeFiles(validFiles);
    }

    return validFiles;
  }, [validateFile]);

  const analyzeFiles = useCallback(async (files) => {
    setIsProcessingFile(true);
    
    const results = [];

    for (const fileObj of files) {
      try {
        // Actualizar estado a "analizando"
        setUploadedFiles((prev) =>
          prev.map((f) =>
            f.id === fileObj.id ? { ...f, status: 'analyzing' } : f
          )
        );

        // Llamar API para an√°lisis con IA
        const formData = new FormData();
        formData.append('file', fileObj.file);
        formData.append('project_id', projectId);

        const response = await onboardingApi.analyzeDocument(projectId, fileObj.file);

        const analysisResult = {
          fileId: fileObj.id,
          fileName: fileObj.name,
          extractedData: response.extracted_data || {},
          categoria: response.categoria,
          subcategoria: response.subcategoria,
          resumen: response.resumen,
          entidades: response.entidades || [],
          confidence: response.confidence || 0,
        };

        results.push(analysisResult);

        // Actualizar estado a "analizado"
        setUploadedFiles((prev) =>
          prev.map((f) =>
            f.id === fileObj.id
              ? { ...f, status: 'analyzed', analysis: analysisResult }
              : f
          )
        );

        // Guardar contenido extra√≠do
        if (response.content) {
          setExtractedDocumentContents((prev) => [
            ...prev,
            { fileId: fileObj.id, content: response.content }
          ]);
        }

      } catch (err) {
        console.error(`Error analizando ${fileObj.name}:`, err);
        
        setUploadedFiles((prev) =>
          prev.map((f) =>
            f.id === fileObj.id
              ? { ...f, status: 'error', error: err.message }
              : f
          )
        );
      }
    }

    setFileAnalysisResults((prev) => [...prev, ...results]);
    setIsProcessingFile(false);

    // Callback con resultados
    if (onAnalysisComplete && results.length > 0) {
      onAnalysisComplete(results);
    }

    return results;
  }, [projectId, onAnalysisComplete]);

  const uploadFilesToCloud = useCallback(async (categoria = null) => {
    const pendingFiles = uploadedFiles.filter(
      (f) => f.status === 'analyzed' || f.status === 'pending'
    );

    if (pendingFiles.length === 0) return [];

    setIsUploadingDocuments(true);
    const uploaded = [];

    try {
      const files = pendingFiles.map((f) => f.file);
      const response = await onboardingApi.uploadFiles(projectId, files, categoria);

      // Actualizar estados
      for (const fileObj of pendingFiles) {
        setUploadedFiles((prev) =>
          prev.map((f) =>
            f.id === fileObj.id ? { ...f, status: 'uploaded' } : f
          )
        );
        uploaded.push(fileObj);
      }

      return response;
    } catch (err) {
      setError(`Error subiendo archivos: ${err.message}`);
      throw err;
    } finally {
      setIsUploadingDocuments(false);
    }
  }, [uploadedFiles, projectId]);

  const removeFile = useCallback((fileId) => {
    setUploadedFiles((prev) => prev.filter((f) => f.id !== fileId));
    setFileAnalysisResults((prev) => prev.filter((r) => r.fileId !== fileId));
    setExtractedDocumentContents((prev) => prev.filter((c) => c.fileId !== fileId));
  }, []);

  const clearAllFiles = useCallback(() => {
    setUploadedFiles([]);
    setAdditionalFiles([]);
    setFileAnalysisResults([]);
    setExtractedDocumentContents([]);
    setError(null);
  }, []);

  const getFileById = useCallback((fileId) => {
    return uploadedFiles.find((f) => f.id === fileId);
  }, [uploadedFiles]);

  const getAnalysisForFile = useCallback((fileId) => {
    return fileAnalysisResults.find((r) => r.fileId === fileId);
  }, [fileAnalysisResults]);

  return {
    // State
    uploadedFiles,
    additionalFiles,
    isProcessingFile,
    isUploadingDocuments,
    fileAnalysisResults,
    extractedDocumentContents,
    uploadProgress,
    error,

    // Actions
    handleFileSelect,
    analyzeFiles,
    uploadFilesToCloud,
    removeFile,
    clearAllFiles,
    getFileById,
    getAnalysisForFile,
    setAdditionalFiles,
    clearError: () => setError(null),
  };
}

export function useOnboardingSteps(initialStep = 0) {
  const [currentStep, setCurrentStep] = useState(initialStep);
  const [collectedData, setCollectedData] = useState({});
  const [entityType, setEntityType] = useState(null); // 'cliente' | 'proveedor'
  const [extractedData, setExtractedData] = useState({});
  const [onboardingStatus, setOnboardingStatus] = useState('idle'); // idle, in_progress, completed, error
  const [onboardingError, setOnboardingError] = useState(null);
  const [showConfirmation, setShowConfirmation] = useState(false);
  const [intelligentMode, setIntelligentMode] = useState(false);
  const [createdEntity, setCreatedEntity] = useState(null);
  const [emailContacto, setEmailContacto] = useState('');

  const currentStepData = ONBOARDING_STEPS[currentStep];

  const validateStep = useCallback((stepIndex = currentStep) => {
    const step = ONBOARDING_STEPS[stepIndex];
    if (!step) return { valid: false, missingFields: [] };

    const missingFields = step.requiredFields.filter((field) => {
      // Verificar en collectedData y extractedData
      const value = collectedData[field] || extractedData[field];
      return !value || (typeof value === 'string' && value.trim() === '');
    });

    return {
      valid: missingFields.length === 0,
      missingFields,
    };
  }, [currentStep, collectedData, extractedData]);

  const nextStep = useCallback(() => {
    const validation = validateStep();
    
    if (!validation.valid) {
      setOnboardingError(`Campos requeridos faltantes: ${validation.missingFields.join(', ')}`);
      return false;
    }

    if (currentStep < ONBOARDING_STEPS.length - 1) {
      setCurrentStep((prev) => prev + 1);
      setOnboardingError(null);
      return true;
    }

    return false;
  }, [currentStep, validateStep]);

  const prevStep = useCallback(() => {
    if (currentStep > 0) {
      setCurrentStep((prev) => prev - 1);
      setOnboardingError(null);
      return true;
    }
    return false;
  }, [currentStep]);

  const goToStep = useCallback((stepIndex) => {
    if (stepIndex >= 0 && stepIndex < ONBOARDING_STEPS.length) {
      setCurrentStep(stepIndex);
      setOnboardingError(null);
      return true;
    }
    return false;
  }, []);

  const updateCollectedData = useCallback((key, value) => {
    setCollectedData((prev) => ({
      ...prev,
      [key]: value,
    }));
  }, []);

  const mergeCollectedData = useCallback((data) => {
    setCollectedData((prev) => ({
      ...prev,
      ...data,
    }));
  }, []);

  const mergeExtractedData = useCallback((data) => {
    setExtractedData((prev) => ({
      ...prev,
      ...data,
    }));
  }, []);

  const selectEntityType = useCallback((type) => {
    if (type === 'cliente' || type === 'proveedor') {
      setEntityType(type);
      updateCollectedData('entityType', type);
      return true;
    }
    return false;
  }, [updateCollectedData]);

  const confirmExtractedData = useCallback(() => {
    // Mergear datos extra√≠dos a datos recolectados
    setCollectedData((prev) => ({
      ...prev,
      ...extractedData,
      dataConfirmed: true,
    }));
    setShowConfirmation(false);
  }, [extractedData]);

  const completeOnboarding = useCallback(async (entityData) => {
    setOnboardingStatus('completed');
    setCreatedEntity(entityData);
    goToStep(ONBOARDING_STEPS.length - 1); // Ir al √∫ltimo paso
  }, [goToStep]);

  const resetOnboarding = useCallback(() => {
    setCurrentStep(0);
    setCollectedData({});
    setEntityType(null);
    setExtractedData({});
    setOnboardingStatus('idle');
    setOnboardingError(null);
    setShowConfirmation(false);

  }, [currentStep]);

    // Solo puede acceder a pasos anteriores o al actual

  }, [currentStep]);

javascriptimport { useState, useCallback, useRef } from 'react';
import apiClient from '../../../api/client';

export function useChatAPI() {
  const [isSearchingWeb, setIsSearchingWeb] = useState(false);
  const [useRealAI, setUseRealAI] = useState(true);
  const [lastError, setLastError] = useState(null);
  const abortControllerRef = useRef(null);

  const sendToClaudeAPI = useCallback(async ({
    message,
    conversationHistory = [],
    agentId = 'ARCHIVO',
    projectId = null,
    empresaId = null,
    contextData = {},
    stream = false,
    onStreamChunk = null,
  }) => {
    // Cancelar request anterior si existe
    if (abortControllerRef.current) {
      abortControllerRef.current.abort();
    }
    abortControllerRef.current = new AbortController();

    setLastError(null);

    try {
      const payload = {
        message,
        conversation_history: conversationHistory,
        agent_id: agentId,
        project_id: projectId,
        empresa_id: empresaId,
        context: contextData,
        stream,
      };

      if (stream && onStreamChunk) {
        // Streaming response
        const response = await fetch('/api/agents/chat/stream', {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json',
            'Authorization': `Bearer ${localStorage.getItem('token')}`,
          },
          body: JSON.stringify(payload),
          signal: abortControllerRef.current.signal,
        });

        if (!response.ok) {
          throw new Error(`HTTP error! status: ${response.status}`);
        }

        const reader = response.body.getReader();
        const decoder = new TextDecoder();
        let fullContent = '';

        while (true) {
          const { done, value } = await reader.read();
          if (done) break;

          const chunk = decoder.decode(value, { stream: true });
          fullContent += chunk;
          onStreamChunk(chunk);
        }

        return {
          message: fullContent,
          agent: agentId,
          streamed: true,
        };
      } else {
        // Regular response
        const response = await apiClient.post('/agents/chat', payload, {
          signal: abortControllerRef.current.signal,
        });

        return {
          message: response.data.message,
          agent: response.data.agent || agentId,
          agentName: response.data.agent_name,
          suggestions: response.data.suggestions || [],
          extractedData: response.data.extracted_data,
          nextAction: response.data.next_action,
          currentPhase: response.data.current_phase,
        };
      }
    } catch (err) {
      if (err.name === 'AbortError') {
        return null; // Cancelled
      }

      const errorMessage = err.response?.data?.detail || err.message || 'Error de conexi√≥n';
      setLastError(errorMessage);
      throw new Error(errorMessage);
    }
  }, []);

  const searchWeb = useCallback(async (query) => {
    setIsSearchingWeb(true);
    setLastError(null);

    try {
      const response = await apiClient.post('/agents/web-search', { query });
      return response.data;
    } catch (err) {
      const errorMessage = err.response?.data?.detail || err.message;
      setLastError(errorMessage);
      throw err;
    } finally {
      setIsSearchingWeb(false);
    }
  }, []);

  const analyzeDocumentWithAI = useCallback(async (file, extractionPrompt = null) => {
    const formData = new FormData();
    formData.append('file', file);
    if (extractionPrompt) {
      formData.append('extraction_prompt', extractionPrompt);
    }

    try {
      const response = await apiClient.post('/agents/analyze-document', formData, {
        headers: { 'Content-Type': 'multipart/form-data' },
      });

      return {
        extractedData: response.data.extracted_data,
        content: response.data.content,
        categoria: response.data.categoria,
        confidence: response.data.confidence,
      };
    } catch (err) {
      const errorMessage = err.response?.data?.detail || err.message;
      setLastError(errorMessage);
      throw err;
    }
  }, []);

  const cancelRequest = useCallback(() => {
    if (abortControllerRef.current) {
      abortControllerRef.current.abort();
    }
  }, []);

  const toggleRealAI = useCallback((enabled) => {
    setUseRealAI(enabled);
  }, []);

  return {
    // State
    isSearchingWeb,
    useRealAI,
    lastError,

    // Actions
    sendToClaudeAPI,
    searchWeb,
    analyzeDocumentWithAI,
    cancelRequest,
    toggleRealAI,
    clearError: () => setLastError(null),
  };
}

jsximport { useState, useRef, useCallback, useEffect } from 'react';

export function ChatInput({
  onSend,
  onAttach,
  disabled = false,
  placeholder = 'Escribe un mensaje...',
  showAttachButton = true,
  maxLength = 4000,
}) {
  const [value, setValue] = useState('');
  const textareaRef = useRef(null);

  // Auto-resize del textarea
  useEffect(() => {
    const textarea = textareaRef.current;
    if (textarea) {
      textarea.style.height = 'auto';
      textarea.style.height = Math.min(textarea.scrollHeight, 200) + 'px';
    }
  }, [value]);

  // Focus al montar
  useEffect(() => {
    if (!disabled) {
      textareaRef.current?.focus();
    }
  }, [disabled]);

  const handleSubmit = useCallback(
    (e) => {
      e?.preventDefault();
      
      const trimmedValue = value.trim();
      if (!trimmedValue || disabled) return;

      onSend(trimmedValue);
      setValue('');
      
      // Reset height
      if (textareaRef.current) {
        textareaRef.current.style.height = 'auto';
      }
    },
    [value, disabled, onSend]
  );

  const handleKeyDown = useCallback(
    (e) => {
      // Enter para enviar, Shift+Enter para nueva l√≠nea
      if (e.key === 'Enter' && !e.shiftKey) {
        e.preventDefault();
        handleSubmit();
      }
    },
    [handleSubmit]
  );

  const handleChange = useCallback(
    (e) => {
      const newValue = e.target.value;
      if (newValue.length <= maxLength) {
        setValue(newValue);
      }
    },
    [maxLength]
  );

  const charactersRemaining = maxLength - value.length;
  const showCharacterCount = value.length > maxLength * 0.8;

  return (
    <form onSubmit={handleSubmit} className="border-t bg-white p-4">
      <div className="flex items-end gap-2">
        {/* Bot√≥n de adjuntar */}
        {showAttachButton && (
          <button
            type="button"
            onClick={onAttach}
            disabled={disabled}
            className="flex-shrink-0 p-2 text-gray-500 hover:text-gray-700 hover:bg-gray-100 rounded-full transition-colors disabled:opacity-50"
            title="Adjuntar archivo"
          >
            <svg
              className="w-6 h-6"
              fill="none"
              viewBox="0 0 24 24"
              stroke="currentColor"
            >
              <path
                strokeLinecap="round"
                strokeLinejoin="round"
                strokeWidth={2}
                d="M15.172 7l-6.586 6.586a2 2 0 102.828 2.828l6.414-6.586a4 4 0 00-5.656-5.656l-6.415 6.585a6 6 0 108.486 8.486L20.5 13"
              />
            </svg>
          </button>
        )}

        {/* Input */}
        <div className="flex-1 relative">
          <textarea
            ref={textareaRef}
            value={value}
            onChange={handleChange}
            onKeyDown={handleKeyDown}
            disabled={disabled}
            placeholder={placeholder}
            rows={1}
            className="w-full px-4 py-3 pr-12 border border-gray-300 rounded-2xl resize-none focus:outline-none focus:ring-2 focus:ring-blue-500 focus:border-transparent disabled:bg-gray-50 disabled:text-gray-500"
            style={{ maxHeight: '200px' }}
          />

          {/* Contador de caracteres */}
          {showCharacterCount && (
            <span
              className={`absolute bottom-1 right-14 text-xs ${
                charactersRemaining < 100 ? 'text-red-500' : 'text-gray-400'
              }`}
            >
              {charactersRemaining}
            </span>
          )}
        </div>

        {/* Bot√≥n enviar */}
        <button
          type="submit"
          disabled={disabled || !value.trim()}
          className="flex-shrink-0 p-3 bg-blue-600 text-white rounded-full hover:bg-blue-700 transition-colors disabled:bg-gray-300 disabled:cursor-not-allowed"
          title="Enviar mensaje"
        >

jsxexport function ConfirmationCard({
  title = 'Confirmar datos',
  data,
  onConfirm,
  onEdit,
  onCancel,
  isLoading = false,
  fieldLabels = {},
}) {
  // Campos a mostrar (excluir internos)
  const displayFields = Object.entries(data || {}).filter(
    ([key]) => !key.startsWith('_') && !['dataConfirmed', 'entityType'].includes(key)
  );

  const getFieldLabel = (key) => {
    return fieldLabels[key] || key.replace(/([A-Z])/g, ' $1').replace(/^./, (s) => s.toUpperCase());
  };

  const formatValue = (value) => {
    if (value === null || value === undefined) return '‚Äî';
    if (typeof value === 'boolean') return value ? 'S√≠' : 'No';
    if (Array.isArray(value)) return value.join(', ');
    if (typeof value === 'object') return JSON.stringify(value);
    return String(value);
  };

  return (
    <div className="bg-white rounded-xl border border-gray-200 shadow-sm overflow-hidden">
      {/* Header */}
      <div className="px-6 py-4 bg-gray-50 border-b">
        <h3 className="text-lg font-semibold text-gray-800">{title}</h3>
        <p className="text-sm text-gray-500 mt-1">
          Revisa que la informaci√≥n sea correcta antes de continuar
        </p>
      </div>

      {/* Campos */}
      <div className="p-6">
        <dl className="space-y-4">
          {displayFields.map(([key, value]) => (
            <div key={key} className="flex flex-col sm:flex-row sm:gap-4">
              <dt className="text-sm font-medium text-gray-500 sm:w-1/3">
                {getFieldLabel(key)}
              </dt>
              <dd className="text-sm text-gray-900 sm:w-2/3 mt-1 sm:mt-0">
                {formatValue(value)}
              </dd>
            </div>
          ))}
        </dl>

        {displayFields.length === 0 && (
          <p className="text-sm text-gray-500 text-center py-4">
            No hay datos para mostrar
          </p>
        )}
      </div>

      {/* Acciones */}
      <div className="px-6 py-4 bg-gray-50 border-t flex flex-col sm:flex-row gap-3 sm:justify-end">
        {onCancel && (
          <button
            onClick={onCancel}
            disabled={isLoading}
            className="px-4 py-2 text-gray-700 bg-white border border-gray-300 rounded-lg hover:bg-gray-50 transition-colors disabled:opacity-50"
          >
            Cancelar
          </button>
        )}
        
        {onEdit && (
          <button
            onClick={onEdit}
            disabled={isLoading}
            className="px-4 py-2 text-blue-600 bg-blue-50 border border-blue-200 rounded-lg hover:bg-blue-100 transition-colors disabled:opacity-50"
          >
            Editar datos
          </button>
        )}
        
        {onConfirm && (
          <button
            onClick={onConfirm}
            disabled={isLoading}
            className="px-4 py-2 text-white bg-blue-600 rounded-lg hover:bg-blue-700 transition-colors disabled:opacity-50 flex items-center gap-2"
          >
            {isLoading && (
              <div className="w-4 h-4 border-2 border-white border-t-transparent rounded-full animate-spin" />
            )}
            Confirmar y continuar
          </button>
        )}
      </div>
    </div>
  );
}

// Constants
import { SYSTEM_MESSAGES, AGENTS } from './constants';

export default function ChatbotOnboarding() {
  const { projectId } = useParams();
  
  // Hooks
  const chat = useChatMessages();
  const api = useChatAPI();
  const files = useFileUpload(projectId, handleFileAnalysisComplete);
  const steps = useOnboardingSteps();

  // Local state
  const [showFileUpload, setShowFileUpload] = useState(false);

  // Mensaje de bienvenida al montar
  useEffect(() => {
    if (chat.messages.length === 0) {
      chat.addBotMessage(SYSTEM_MESSAGES.WELCOME, {
        agent: 'ARCHIVO',
        suggestions: [
          { text: 'Registrar cliente', value: 'cliente' },
          { text: 'Registrar proveedor', value: 'proveedor' },
        ],
      });
    }
  }, []);

  // Handler para enviar mensaje
  const handleSendMessage = useCallback(async (content) => {
    // Agregar mensaje del usuario
    chat.addUserMessage(content);
    chat.setTyping(true);

    try {
      // Enviar a la API
      const response = await api.sendToClaudeAPI({
        message: content,
        conversationHistory: chat.conversationHistory,
        agentId: steps.currentStepData?.agent || 'ARCHIVO',
        projectId,
        contextData: {
          currentStep: steps.currentStep,
          entityType: steps.entityType,
          collectedData: steps.collectedData,
          extractedData: steps.extractedData,
        },
      });

      chat.setTyping(false);

      if (response) {
        // Agregar respuesta del bot
        chat.addBotMessage(response.message, {
          agent: response.agent,
          agentName: response.agentName,
          suggestions: response.suggestions,
        });

        // Procesar datos extra√≠dos
        if (response.extractedData) {
          steps.mergeExtractedData(response.extractedData);
        }

        // Procesar acciones
        if (response.nextAction) {
          handleBotAction(response.nextAction);
        }
      }
    } catch (err) {
      chat.setTyping(false);
      chat.addSystemMessage(`Error: ${err.message}`, 'error');
    }
  }, [chat, api, steps, projectId]);

  // Handler para acciones del bot
  const handleBotAction = useCallback((action) => {
    switch (action.type) {
      case 'select_entity_type':
        steps.selectEntityType(action.value);
        steps.nextStep();
        break;
      case 'request_documents':
        setShowFileUpload(true);
        break;
      case 'show_confirmation':
        steps.setShowConfirmation(true);
        break;
      case 'complete_onboarding':
        steps.completeOnboarding(action.data);
        break;
      default:
        console.log('Acci√≥n no manejada:', action);
    }
  }, [steps]);

  // Handler para sugerencias clickeadas
  const handleSuggestionClick = useCallback((suggestion) => {
    const value = typeof suggestion === 'string' ? suggestion : suggestion.value || suggestion.text;
    handleSendMessage(value);
  }, [handleSendMessage]);

  // Handler para an√°lisis de archivos completado
  function handleFileAnalysisComplete(results) {
    // Notificar en el chat
    chat.addSystemMessage(SYSTEM_MESSAGES.ANALYSIS_COMPLETE, 'success');

    // Mergear datos extra√≠dos
    results.forEach((result) => {
      if (result.extractedData) {
        steps.mergeExtractedData(result.extractedData);
      }
    });

    // Mostrar confirmaci√≥n
    if (Object.keys(steps.extractedData).length > 0) {
      steps.setShowConfirmation(true);
    }
  }

  // Handler para confirmar datos
  const handleConfirmData = useCallback(async () => {
    steps.confirmExtractedData();
    steps.nextStep();

    chat.addBotMessage('¬°Perfecto! Los datos han sido confirmados. Continuemos con el siguiente paso.', {
      agent: 'ARCHIVO',
    });
  }, [steps, chat]);

  return (
    <div className="flex h-screen bg-gray-100">
      {/* Sidebar con progreso */}
      <aside className="hidden lg:block w-80 bg-white border-r overflow-y-auto">
        <div className="p-4">
          <h2 className="text-xl font-bold text-gray-800 mb-2">
            Onboarding
          </h2>
          <p className="text-sm text-gray-500 mb-6">
            {steps.entityType === 'cliente' ? 'Nuevo Cliente' : 
             steps.entityType === 'proveedor' ? 'Nuevo Proveedor' : 
             'Selecciona el tipo de entidad'}
          </p>

          <ProgressIndicator
            steps={steps.steps}
            currentStep={steps.currentStep}
            onStepClick={steps.goToStep}
          />
        </div>
      </aside>

      {/* Chat principal */}
      <main className="flex-1 flex flex-col">
        {/* Header m√≥vil */}
        <header className="lg:hidden bg-white border-b px-4 py-3 flex items-center justify-between">
          <h1 className="font-semibold text-gray-800">Onboarding</h1>
          <span className="text-sm text-gray-500">
            Paso {steps.currentStep + 1} de {steps.steps.length}
          </span>
        </header>

        {/* Mensajes */}
        <MessageList
          messages={chat.messages}
          isTyping={chat.isTyping}
          currentAgent={steps.currentStepData?.agent}
          onSuggestionClick={handleSuggestionClick}
          className="flex-1"
        />

        {/* Error */}
        {(api.lastError || steps.onboardingError) && (
          <div className="px-4">
            <ErrorMessage
              message={api.lastError || steps.onboardingError}
              onRetry={() => {
                api.clearError();
                steps.clearError();
              }}
            />
          </div>
        )}

        {/* Input */}
        <ChatInput
          onSend={handleSendMessage}
          onAttach={() => setShowFileUpload(true)}
          disabled={chat.isTyping || api.isSearchingWeb}
          placeholder={
            steps.entityType
              ? `Escribe aqu√≠ para continuar con el registro del ${steps.entityType}...`
              : 'Escribe aqu√≠ para comenzar...'
          }
        />

      <Modal

        title="Subir documentos"
        size="lg"
      >

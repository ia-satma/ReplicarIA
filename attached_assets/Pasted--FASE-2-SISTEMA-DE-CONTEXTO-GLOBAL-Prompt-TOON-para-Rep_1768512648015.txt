      requiere_revision_si: ["monto >= umbral", "risk_score >= 60", "tipologia en siempre_humano", "parte_relacionada"]
    },

function requiereRevisionHumana(proyecto, riskScore, proveedor) {
  
  // Por monto
  if (proyecto.monto >= UMBRALES_REVISION_HUMANA.por_monto.umbral_obligatorio) {
    return {
      requiere: true,
      razon: "Monto >= " + UMBRALES_REVISION_HUMANA.por_monto.umbral_obligatorio + " MXN"
    };
  }
  
  // Por risk_score
  if (riskScore >= UMBRALES_REVISION_HUMANA.por_risk_score.obligatorio.umbral) {
    return {
      requiere: true,
      razon: "Risk score >= " + UMBRALES_REVISION_HUMANA.por_risk_score.obligatorio.umbral
    };
  }
  
  // Por tipología
  if (UMBRALES_REVISION_HUMANA.por_tipologia.siempre_humano.includes(proyecto.tipologia)) {
    return {
      requiere: true,
      razon: "Tipología " + proyecto.tipologia + " requiere revisión humana obligatoria"
    };
  }
  
  // Por proveedor
  if (proveedor.tipo_relacion !== 'TERCERO_INDEPENDIENTE') {
    return {
      requiere: true,
      razon: "Operación con parte relacionada: " + proveedor.tipo_relacion
    };
  }
  
  if (proveedor.alerta_efos) {
    return {
      requiere: true,
      razon: "Proveedor con alerta EFOS activa"
    };
  }
  
  // No requiere
  return {
    requiere: false,
    razon: null
  };
}

module.exports = { UMBRALES_REVISION_HUMANA, requiereRevisionHumana };

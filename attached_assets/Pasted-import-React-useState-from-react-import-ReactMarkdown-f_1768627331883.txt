import React, { useState } from 'react';
import ReactMarkdown from 'react-markdown';

/**
 * Componente para mostrar reportes narrativos humanizados
 */
const ReporteNarrativo = ({ reporte, tipo = 'markdown' }) => {
  const [vistaActiva, setVistaActiva] = useState(tipo);
  
  if (!reporte) {
    return (
      <div className="p-4 bg-gray-50 rounded-lg text-gray-500">
        No hay reporte disponible
      </div>
    );
  }
  
  const { markdown, html, persona } = reporte;
  
  return (
    <div className="reporte-narrativo bg-white rounded-lg shadow-lg overflow-hidden">
      {/* Header con informaciÃ³n del autor */}
      <div className="bg-gradient-to-r from-blue-800 to-blue-600 text-white p-4">
        <div className="flex items-center justify-between">
          <div>
            <h3 className="text-lg font-semibold">{persona?.nombre || 'Analista'}</h3>
            <p className="text-blue-200 text-sm">{persona?.titulo || 'Especialista'}</p>
          </div>
          
          {/* Selector de vista */}
          <div className="flex gap-2">
            <button
              onClick={() => setVistaActiva('markdown')}
              className={`px-3 py-1 rounded text-sm ${
                vistaActiva === 'markdown' 
                  ? 'bg-white text-blue-800' 
                  : 'bg-blue-700 text-white hover:bg-blue-600'
              }`}
            >
              ğŸ“„ Texto
            </button>
            <button
              onClick={() => setVistaActiva('html')}
              className={`px-3 py-1 rounded text-sm ${
                vistaActiva === 'html' 
                  ? 'bg-white text-blue-800' 
                  : 'bg-blue-700 text-white hover:bg-blue-600'
              }`}
            >
              ğŸ¨ Formateado
            </button>
          </div>
        </div>
      </div>
      
      {/* Contenido del reporte */}
      <div className="p-6 max-h-[600px] overflow-y-auto">
        {vistaActiva === 'markdown' ? (
          <div className="prose prose-blue max-w-none">
            <ReactMarkdown>{markdown}</ReactMarkdown>
          </div>
        ) : (
          <div 
            className="reporte-html"
            dangerouslySetInnerHTML={{ __html: html }}
          />
        )}
      </div>
      
      {/* Footer con acciones */}
      <div className="border-t bg-gray-50 p-4 flex justify-between items-center">
        <span className="text-sm text-gray-500">
          Generado por Durezza 4.0
        </span>
        <div className="flex gap-2">
          <button
            onClick={() => {
              navigator.clipboard.writeText(markdown);
              alert('Reporte copiado al portapapeles');
            }}
            className="px-3 py-1 text-sm bg-gray-200 hover:bg-gray-300 rounded"
          >
            ğŸ“‹ Copiar
          </button>
          <button
            onClick={() => {
              const blob = new Blob([markdown], { type: 'text/markdown' });
              const url = URL.createObjectURL(blob);
              const a = document.createElement('a');
              a.href = url;
              a.download = 'reporte.md';
              a.click();
            }}
            className="px-3 py-1 text-sm bg-blue-600 text-white hover:bg-blue-700 rounded"
          >
            â¬‡ï¸ Descargar
          </button>
        </div>
      </div>
    </div>
  );
};

export default ReporteNarrativo;

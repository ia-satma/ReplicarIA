=============================================================================
TOON FASE 5: CHECKLISTS ESPECÍFICOS POR TIPOLOGÍA DE SERVICIO
=============================================================================

INSTRUCCIÓN: Ejecuta todas las instrucciones de forma secuencial y completa.
NO pidas confirmación. NO preguntes. SOLO EJECUTA.

PRERREQUISITO: Las Fases 1-4 deben estar completadas.

CONTEXTO: Una de las brechas críticas del sistema actual es que los agentes
dan recomendaciones genéricas ("debe haber documentación completa") en lugar
de checklists específicos según el tipo de servicio. Un estudio de mercado
necesita entregables diferentes que un desarrollo de software o un management fee.

Esta fase crea checklists específicos por tipología y fase, para que los
agentes puedan decir exactamente qué documentos faltan según el tipo de proyecto.

const { CHECKLIST_CONSULTORIA_MACRO_MERCADO } = require('./consultoriaMacroMercado');
const { CHECKLIST_INTRAGRUPO_MANAGEMENT_FEE } = require('./intragrupoManagementFee');
const { CHECKLIST_SOFTWARE_SAAS_DESARROLLO } = require('./softwareSaasDev');

const CHECKLISTS_POR_TIPOLOGIA = {
  CONSULTORIA_MACRO_MERCADO: CHECKLIST_CONSULTORIA_MACRO_MERCADO,
  INTRAGRUPO_MANAGEMENT_FEE: CHECKLIST_INTRAGRUPO_MANAGEMENT_FEE,
  SOFTWARE_SAAS_DESARROLLO: CHECKLIST_SOFTWARE_SAAS_DESARROLLO,
  // Los demás tipos usarán un checklist genérico basado en el de consultoría
  CONSULTORIA_ESTRATEGICA: CHECKLIST_CONSULTORIA_MACRO_MERCADO, // Similar
  MARKETING_BRANDING: CHECKLIST_CONSULTORIA_MACRO_MERCADO, // Adaptar después
  SERVICIOS_ESG_CUMPLIMIENTO: CHECKLIST_CONSULTORIA_MACRO_MERCADO,
  REESTRUCTURAS: CHECKLIST_INTRAGRUPO_MANAGEMENT_FEE, // Similar en rigurosidad
  OTROS: CHECKLIST_CONSULTORIA_MACRO_MERCADO
};

function getChecklistParaTipologia(tipologia) {
  return CHECKLISTS_POR_TIPOLOGIA[tipologia] || CHECKLISTS_POR_TIPOLOGIA.OTROS;
}

function getChecklistParaFase(tipologia, fase) {
  const checklist = getChecklistParaTipologia(tipologia);
  return checklist[fase] || null;
}

function getItemsObligatorios(tipologia, fase) {
  const checklistFase = getChecklistParaFase(tipologia, fase);
  if (!checklistFase || !checklistFase.items) return [];
  return checklistFase.items.filter(item => item.obligatorio === true);
}

function getItemsPendientes(tipologia, fase, itemsEntregados) {
  const checklistFase = getChecklistParaFase(tipologia, fase);
  if (!checklistFase || !checklistFase.items) return [];
  
  const idsEntregados = itemsEntregados.map(i => i.id);
  return checklistFase.items.filter(item => !idsEntregados.includes(item.id));
}

module.exports = {
  CHECKLISTS_POR_TIPOLOGIA,
  getChecklistParaTipologia,
  getChecklistParaFase,
  getItemsObligatorios,
  getItemsPendientes,
  CHECKLIST_CONSULTORIA_MACRO_MERCADO,
  CHECKLIST_INTRAGRUPO_MANAGEMENT_FEE,
  CHECKLIST_SOFTWARE_SAAS_DESARROLLO
};

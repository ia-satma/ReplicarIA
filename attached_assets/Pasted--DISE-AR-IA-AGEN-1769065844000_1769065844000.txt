        inicio = datetime.now()

        resultados = {

        }
        
        # Calcular puntuaciÃ³n general
        scores = [r.get('score', 0) for r in resultados.values() if isinstance(r, dict)]

        resultados['tiempo_auditoria_ms'] = (datetime.now() - inicio).total_seconds() * 1000

    def auditar_colores(self) -> Dict[str, Any]:
        """Analiza uso de colores en el cÃ³digo"""
        
        colores_encontrados = {}
        colores_prohibidos_usados = []
        archivos_analizados = 0
        
        # Patrones para detectar colores
        patrones = [
            r'#[0-9A-Fa-f]{6}\b',  # Hex de 6 dÃ­gitos
            r'#[0-9A-Fa-f]{3}\b',  # Hex de 3 dÃ­gitos
            r'rgb\([^)]+\)',       # RGB
            r'rgba\([^)]+\)',      # RGBA
            r'(?:bg|text|border|fill|stroke)-(?:red|green|blue|yellow|purple|pink|orange)-\d+',  # Tailwind
        ]
        
        # Colores de la paleta oficial (para verificar uso)
        colores_oficiales = self._extraer_colores_oficiales()
        
        for root, dirs, files in os.walk(self.ruta_frontend):
            # Ignorar node_modules
            dirs[:] = [d for d in dirs if d != 'node_modules']
            
            for file in files:
                if file.endswith(('.jsx', '.tsx', '.js', '.ts', '.css', '.scss')):
                    archivos_analizados += 1
                    filepath = os.path.join(root, file)
                    
                    try:
                        with open(filepath, 'r', encoding='utf-8') as f:
                            contenido = f.read()
                            
                            for patron in patrones:
                                matches = re.findall(patron, contenido, re.IGNORECASE)
                                for color in matches:
                                    color_normalizado = color.upper() if color.startswith('#') else color
                                    
                                    if color_normalizado not in colores_encontrados:
                                        colores_encontrados[color_normalizado] = {
                                            'usos': 0,
                                            'archivos': []
                                        }
                                    
                                    colores_encontrados[color_normalizado]['usos'] += 1
                                    if filepath not in colores_encontrados[color_normalizado]['archivos']:
                                        colores_encontrados[color_normalizado]['archivos'].append(
                                            filepath.replace(self.ruta_frontend + '/', '')
                                        )
                                    
                                    # Verificar si es prohibido
                                    if color_normalizado in COLORES_PROHIBIDOS or color.lower() in COLORES_PROHIBIDOS:
                                        if color_normalizado not in [c['color'] for c in colores_prohibidos_usados]:
                                            colores_prohibidos_usados.append({
                                                'color': color_normalizado,
                                                'archivo': filepath.replace(self.ruta_frontend + '/', ''),
                                                'sugerencia': self._sugerir_color_alternativo(color_normalizado)
                                            })
                                            
                                            self.problemas.append({
                                                'tipo': 'color_prohibido',
                                                'severidad': 'media',
                                                'mensaje': f'Color prohibido {color_normalizado} en {file}',
                                                'archivo': filepath,
                                                'sugerencia': self._sugerir_color_alternativo(color_normalizado)
                                            })
                    except Exception as e:
                        pass
        
        # Calcular score
        total_colores = len(colores_encontrados)
        colores_fuera = len(colores_prohibidos_usados)
        score = max(0, 100 - (colores_fuera * 10))
        
        return {
            'score': score,
            'total_colores': total_colores,
            'colores_oficiales_usados': len([c for c in colores_encontrados if c in colores_oficiales]),
            'colores_prohibidos': colores_prohibidos_usados,
            'archivos_analizados': archivos_analizados,
            'detalle': {k: v for k, v in list(colores_encontrados.items())[:20]}  # Top 20
        }
    
    def _extraer_colores_oficiales(self) -> List[str]:
        """Extrae lista de colores oficiales del design system"""
        colores = []

            if isinstance(obj, dict):
                for k, v in obj.items():
                    if isinstance(v, str) and v.startswith('#'):

                    else:
                        extraer_recursivo(v)

    def auditar_responsive(self) -> Dict[str, Any]:
        """Analiza optimizaciÃ³n mÃ³vil"""
        
        componentes_analizados = 0
        problemas_responsive = []
        
        # Patrones que indican problemas de responsive
        patrones_problematicos = [
            (r'width:\s*\d{4,}px', 'Ancho fijo muy grande'),
            (r'height:\s*\d{3,}px', 'Alto fijo puede causar overflow'),
            (r'font-size:\s*\d{1,2}px', 'TamaÃ±o de fuente fijo pequeÃ±o'),
            (r'overflow:\s*hidden', 'Overflow hidden puede ocultar contenido'),
        ]
        
        # Patrones que indican buenas prÃ¡cticas
        patrones_buenos = [
            r'@media',
            r'sm:|md:|lg:|xl:',  # Breakpoints de Tailwind
            r'flex|grid',
            r'max-w-|min-w-',
            r'responsive',
        ]
        
        for root, dirs, files in os.walk(self.ruta_frontend):
            dirs[:] = [d for d in dirs if d != 'node_modules']
            
            for file in files:
                if file.endswith(('.jsx', '.tsx')):
                    componentes_analizados += 1
                    filepath = os.path.join(root, file)
                    
                    try:
                        with open(filepath, 'r', encoding='utf-8') as f:
                            contenido = f.read()
                            
                            # Buscar problemas
                            for patron, descripcion in patrones_problematicos:
                                if re.search(patron, contenido):
                                    problemas_responsive.append({
                                        'archivo': file,
                                        'problema': descripcion,
                                        'severidad': 'media'
                                    })
                            
                            # Verificar si usa breakpoints
                            usa_responsive = any(re.search(p, contenido) for p in patrones_buenos)
                            
                            if not usa_responsive and len(contenido) > 500:
                                problemas_responsive.append({
                                    'archivo': file,
                                    'problema': 'No usa breakpoints responsive',
                                    'severidad': 'baja'
                                })
                    except:
                        pass
        
        # Calcular score
        score = max(0, 100 - len(problemas_responsive) * 5)
        
        return {
            'score': score,
            'componentes_analizados': componentes_analizados,
            'problemas': problemas_responsive[:10],  # Top 10
            'total_problemas': len(problemas_responsive)
        }

import React, { useState } from 'react';

function CargaDocumentos({ proyectoId, faseActual, onDocumentoCargado }) {
  const [archivo, setArchivo] = useState(null);
  const [tipo, setTipo] = useState('');
  const [descripcion, setDescripcion] = useState('');
  const [uploading, setUploading] = useState(false);

  const tiposDocumento = [
    { value: 'SIB_BEE', label: 'SIB / BEE' },
    { value: 'SOW', label: 'SOW / Orden de Trabajo' },
    { value: 'CONTRATO', label: 'Contrato' },
    { value: 'ANEXO_TECNICO', label: 'Anexo T√©cnico' },
    { value: 'INFORME', label: 'Informe / Entregable' },
    { value: 'MODELO_PARAMETRICO', label: 'Modelo Param√©trico' },
    { value: 'DASHBOARD', label: 'Dashboard' },
    { value: 'MANUAL', label: 'Manual / Documentaci√≥n' },
    { value: 'MINUTA', label: 'Minuta de Reuni√≥n' },
    { value: 'CRONOGRAMA', label: 'Cronograma' },
    { value: 'CFDI', label: 'CFDI / Factura' },
    { value: 'COMPROBANTE_PAGO', label: 'Comprobante de Pago' },
    { value: 'EVIDENCIA_EJECUCION', label: 'Evidencia de Ejecuci√≥n' },
    { value: 'OTRO', label: 'Otro' }
  ];

  async function handleUpload(e) {
    e.preventDefault();
    if (!archivo || !tipo) {
      alert('Selecciona un archivo y tipo de documento');
      return;
    }

    setUploading(true);
    try {
      const formData = new FormData();
      formData.append('archivo', archivo);
      formData.append('tipo', tipo);
      formData.append('descripcion', descripcion);
      formData.append('fase_asociada', faseActual);

      const response = await fetch(`/api/proyectos/${proyectoId}/documentos`, {
        method: 'POST',
        body: formData
      });

      const data = await response.json();

      if (response.ok) {
        alert('‚úÖ Documento cargado exitosamente');
        setArchivo(null);
        setTipo('');
        setDescripcion('');
        if (onDocumentoCargado) onDocumentoCargado(data);
      } else {
        alert('‚ùå Error: ' + data.error);
      }
    } catch (error) {
      alert('Error: ' + error.message);
    }
    setUploading(false);
  }

  return (
    <div className="bg-white p-6 rounded-lg shadow">
      <h3 className="font-bold text-lg mb-4">Cargar Documento</h3>
      
      <form onSubmit={handleUpload} className="space-y-4">
        <div>
          <label className="block text-sm font-medium mb-1">Tipo de documento</label>
          <select
            value={tipo}
            onChange={(e) => setTipo(e.target.value)}
            className="w-full border rounded px-3 py-2"
            required
          >
            <option value="">Seleccionar tipo...</option>
            {tiposDocumento.map(t => (
              <option key={t.value} value={t.value}>{t.label}</option>
            ))}
          </select>
        </div>

        <div>
          <label className="block text-sm font-medium mb-1">Descripci√≥n</label>
          <input
            type="text"
            value={descripcion}
            onChange={(e) => setDescripcion(e.target.value)}
            className="w-full border rounded px-3 py-2"
            placeholder="Ej: Informe Final V2.0 - Estudio de Mercado NL"
          />
        </div>

        <div>
          <label className="block text-sm font-medium mb-1">Archivo</label>
          <input
            type="file"
            onChange={(e) => setArchivo(e.target.files[0])}
            className="w-full border rounded px-3 py-2"
            required

        </div>

import React from 'react';

function TimelineFases({ faseActual, fasesCompletadas = [] }) {
  const fases = [
    { id: 'F0', nombre: 'Aprobaci√≥n BEE', candado: false },
    { id: 'F1', nombre: 'Pre-contrataci√≥n', candado: false },
    { id: 'F2', nombre: 'Validaci√≥n inicio', candado: true },
    { id: 'F3', nombre: 'Ejecuci√≥n inicial', candado: false },
    { id: 'F4', nombre: 'Revisi√≥n iterativa', candado: false },
    { id: 'F5', nombre: 'Entrega final', candado: false },
    { id: 'F6', nombre: 'VBC Fiscal/Legal', candado: true },
    { id: 'F7', nombre: 'Auditor√≠a interna', candado: false },
    { id: 'F8', nombre: 'CFDI y pago', candado: true },
    { id: 'F9', nombre: 'Post-implementaci√≥n', candado: false }
  ];

  function getEstadoFase(faseId) {
    if (fasesCompletadas.includes(faseId)) return 'completada';
    if (faseId === faseActual) return 'actual';
    const indexActual = fases.findIndex(f => f.id === faseActual);
    const indexFase = fases.findIndex(f => f.id === faseId);
    if (indexFase < indexActual) return 'completada';
    return 'pendiente';
  }

  return (
    <div className="bg-white p-6 rounded-lg shadow">
      <h3 className="font-bold text-lg mb-4">Progreso del Proyecto</h3>
      
      <div className="relative">
        {/* L√≠nea de progreso */}
        <div className="absolute top-5 left-0 right-0 h-1 bg-gray-200" />
        <div 
          className="absolute top-5 left-0 h-1 bg-blue-500 transition-all"
          style={{ 
            width: `${(fases.findIndex(f => f.id === faseActual) / (fases.length - 1)) * 100}%` 
          }}
        />
        
        {/* Nodos de fases */}
        <div className="relative flex justify-between">
          {fases.map((fase, index) => {
            const estado = getEstadoFase(fase.id);
            return (
              <div key={fase.id} className="flex flex-col items-center" style={{ width: '10%' }}>
                <div className={`
                  w-10 h-10 rounded-full flex items-center justify-center text-sm font-bold
                  ${estado === 'completada' ? 'bg-green-500 text-white' :
                    estado === 'actual' ? 'bg-blue-500 text-white ring-4 ring-blue-200' :
                    'bg-gray-200 text-gray-500'}
                `}>
                  {fase.candado && estado !== 'completada' ? 'üîí' : fase.id}
                </div>
                <div className="mt-2 text-xs text-center text-gray-600">
                  {fase.nombre}
                </div>
                {fase.candado && (
                  <div className="text-xs text-red-500 mt-1">Candado</div>
                )}
              </div>
            );
          })}
        </div>
      </div>
    </div>
  );
}
